# AES数据加密

 

### 详细代码
- 安装命令 ```pnpm i crypto-js```
```javascript 
import { decrypt, encrypt } from 'crypto-js/aes'; 
import UTF8 from 'crypto-js/enc-utf8'; 
import ECB from 'crypto-js/mode-ecb'; 
import pkcs7 from 'crypto-js/pad-pkcs7';
import { cacheCipher } from '@/config';
export const cacheCipher = { key: '_11111000001111@', iv: '@11111000001111_', }; 
export const getOptions: any = () => { return { mode: ECB, padding: pkcs7, iv: cacheCipher.iv, }; }; 
// encryption 
export const encryptByAES = (cipherText: string) => { return encrypt(cipherText, cacheCipher.key, getOptions()).toString(); };
// decrypt 
export const decryptByAES = (cipherText: string) => { return decrypt(cipherText, cacheCipher.key, getOptions()).toString(UTF8); }; 
```
